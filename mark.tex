    % For a point on an ellipse, we have $x = a \cos\theta$ and $y = b
    % \cos\theta$.  That means I just need to pick an angle at which the
    % ellipse looks good.  Note, 45 is equivalent to using the fit
    % library which is a bit too narrow for my taste (today).
    \def\ang{55}
    \draw [ultra thick]
        let
            \p1 = (p1),
            \n1 = {\x1 / cos(\ang)},
            \n2 = {\y1 / sin(\ang)}
        in
            circle [x radius=\n1, y radius=\n2];
